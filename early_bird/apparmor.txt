#include <tunables/global>

profile early_bird flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal,
  network inet tcp,
  network inet udp,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /app/run.py cx -> early_bird_app,

  profile early_bird_app flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal receive,

    # Python
    /usr/bin/python3 ix,
    /usr/lib/python3*/** mr,

    # Access to addon files
    /app/** r,
    /app/templates/** r,
    /app/translations/** r,
    
    # Data storage
    /data/** rw,

    # Options.json
    /data/options.json r,

    # Networking for Flask
    network inet tcp,
    network inet udp,
  }
}
